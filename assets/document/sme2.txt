        function changeTableBenefit(){
            var choice = $('#BenefitChoiceSelect').val();
            switch(choice){
                case 'IP':
                    $('.table-ip').removeClass('hide');
                    $('.table-op').addClass('hide');
                    $('.table-d').addClass('hide');
                    $('.table-m').addClass('hide');
                    $('.table-o').addClass('hide');
                break;
                case 'OP':
                    $('.table-ip').addClass('hide');
                    $('.table-op').removeClass('hide');
                    $('.table-d').addClass('hide');
                    $('.table-m').addClass('hide');
                    $('.table-o').addClass('hide');
                break;
                case 'D':
                    $('.table-ip').addClass('hide');
                    $('.table-op').addClass('hide');
                    $('.table-d').removeClass('hide');
                    $('.table-m').addClass('hide');
                    $('.table-o').addClass('hide');
                break;
                case 'M':
                    $('.table-ip').addClass('hide');
                    $('.table-op').addClass('hide');
                    $('.table-d').addClass('hide');
                    $('.table-m').removeClass('hide');
                    $('.table-o').addClass('hide');
                break;
                case 'O':
                    $('.table-ip').addClass('hide');
                    $('.table-op').addClass('hide');
                    $('.table-d').addClass('hide');
                    $('.table-m').addClass('hide');
                    $('.table-o').removeClass('hide');
                break;
                default:
                    $('.table-ip').addClass('hide');
                    $('.table-op').addClass('hide');
                    $('.table-d').addClass('hide');
                    $('.table-m').addClass('hide');
                    $('.table-o').addClass('hide');
                break;
            }
            detachStickyKit();
            activeStickyKit();

        }
